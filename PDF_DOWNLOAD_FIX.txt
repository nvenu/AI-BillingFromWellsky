            const filename = `paper-claim-${sanitizedInsurance}-${timestamp}.pdf`;

            // Download the PDF
            console.log(`  Downloading PDF...`);
            try {
              // Check if there's an embed or iframe with the actual PDF URL
              const pdfSrc = await newPage.evaluate(() => {
                const embed = document.querySelector('embed[type="application/pdf"]');
                if (embed) return (embed as HTMLEmbedElement).src;
                
                const iframe = document.querySelector('iframe');
                if (iframe) return (iframe as HTMLIFrameElement).src;
                
                const object = document.querySelector('object[type="application/pdf"]');
                if (object) return (object as HTMLObjectElement).data;
                
                return null;
              });
              
              console.log(`  PDF source found: ${pdfSrc || 'none, using page URL'}`);
              
              // Use the actual PDF URL if found, otherwise use page URL
              const downloadUrl = pdfSrc || pdfUrl;
              
              // Download using fetch through the browser context
              const response = await newPage.context().request.get(downloadUrl);
              const pdfBuffer = await response.body();
              
              // Save the PDF
              fs.writeFileSync(filepath, pdfBuffer);
              downloadedFiles.push(filepath);
              console.log(`  ✓ Downloaded: ${filename} (${pdfBuffer.length} bytes)`);
            } catch (pdfError: any) {

              try {
                const pdfBuffer = await newPage.pdf({
                  format: 'Letter',

                  preferCSSPageSize: true
                });

                console.error(`  ✗ Fallback also failed:`, fallbackError?.message || fallbackError);
              }
            }
